 18    def get_chemical(self, op :str, word: str, **kwargs) -> Dict[str, Any]:
 19        """
 20        #### Description:
 21            Fetch chemical details based starting characters, exact match or substring of search word.
 22

 26
 27        #### Query Parameters:
 28            - top: Int32  -> Number of records to return.

 32
 33        #### Output Schema:
 34            {
 35                "casrn": "string",
 36                "dtxsid": "string",
 37                "dtxcid": "string",
 38                "preferredName": "string",
 39                "hasStructureImage": 0,
 40                "smiles": "string",
 41                "isMarkush": false,
 42                "searchName": "string",
 43                "searchValue": "string",
 44                "rank": 0
 45            }
 46
 47        #### Example:
 48            client = ChemSearch(api_key=api_key)
 49            kawrgs = {"params": {"top": 20}}
 50            response = client.get_chemical(op="contain", word="95-16-9")
 51
 52            kwargs = {"params": {"top": 20, "projection": "chemicalsearchall"}
 53            response = client.get_chemical(op="contain", word="95-16-9", **kwargs)
 54
 55        """
 56        op = op.lower()
 57        if op not in ["start-with", "equal", "contain"]:
 58            raise ValueError("Invalid operator. Valid values are 'start-with', 'equal', 'contain'")

 61        elif op == "equal":
 62            resource_id = f"chemical/search/equal/{word}"
 63        elif op == "contain":
 64            resource_id = f"chemical/search/contain/{word}"
